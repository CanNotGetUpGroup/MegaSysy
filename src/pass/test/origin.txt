define dso_local i32 @ifElseIf(){

9:     ; else Stmt

12:

  %14= icmp eq i32 %13, 1     ; b==10 && a==1
  br i1 %14, label %15, label %16     ; judge b==10&&a==1

15:     ; if Stmt

16:     ; else Stmt
  %17 = load i32, i32* %0

  %21= icmp eq i32 %20, -5     ; b==10 && a==-5

  %24 = add i32 %23, 15     ; a+15

  store i32 %27, i32* %1     ; a=-+a
  br label %28     

28:     
  br label %29     

29:     
  br label %30     

30:     
  %31 = load i32, i32* %1     
  ret i32 %31     
}

define dso_local i32 @main(){
